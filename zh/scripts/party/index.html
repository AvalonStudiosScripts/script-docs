    avalon_party

    FiveM MMORPG 风格组队系统。创建带有实时 HUD 的队伍，自定义生命值条、准备检查、共享路点、自动观战模式和目标标记系统。

    版本： 1.0.0 | 作者： Avalon | 框架： ESX / QBCore / QBox / Standalone

    依赖项

    Resource 是否必须
    ox_lib ✅
    ox_target ✅

    命令

    命令 参数 说明
    /partyinvite [玩家ID] 邀请玩家入队
    /partyaccept — 接受邀请
    /partydecline — 拒绝邀请
    /partyleave — 离开队伍
    /partydisband — 解散队伍（队长）
    /partykick [玩家ID] 踢出成员（队长）
    /partytransfer [玩家ID] 转让队长（队长）
    /partyready — 发起准备检查（队长）
    /partywp — 共享路点（队长）

    生命值条配置

    ID 标签 自动更新
    health HP ✅
    stamina 耐力 ✅
    mana 魔法 ❌ 需要 SetMemberBar
    shield 护盾 ❌
    rage 怒气 ❌

    可用导出函数

    服务端

    IsInParty(source) → boolean

    if exports['avalon_party']:IsInParty(source) then
    -- 玩家在队伍中
end

    GetPartyMembers(source) → table | nil

    获取队伍所有成员的服务器ID列表。 

    local members = exports['avalon_party']:GetPartyMembers(source)
if members then
    for _, id in ipairs(members) do
        exports.ox_inventory:AddItem(id, 'health_potion', 1)
    end
end

    GetPartyLeader(source) → number | nil

    local leader = exports['avalon_party']:GetPartyLeader(source)

    SetMemberBar(source, barId, value, maxValue) → boolean

    更新队伍 HUD 中玩家的自定义条，所有成员实时可见。

    -- 施法后更新魔法条
RegisterNetEvent('magic:cast', function(cost)
    local newMana = math.max(0, GetMana(source) - cost)
    exports['avalon_party']:SetMemberBar(source, 'mana', newMana, 100)
end)

-- 受伤后更新护盾条
RegisterNetEvent('combat:damage', function(dmg)
    local newShield = math.max(0, GetShield(source) - dmg)
    exports['avalon_party']:SetMemberBar(source, 'shield', newShield, 100)
end)

    SetMemberStatus(source, statusId, active) → boolean

    在成员卡片上显示/隐藏状态图标。

    -- 显示"中毒"
exports['avalon_party']:SetMemberStatus(source, 'poisoned', true)
-- 移除"中毒"
exports['avalon_party']:SetMemberStatus(source, 'poisoned', false)

-- 战斗时显示战斗状态，5秒后自动消失
exports['avalon_party']:SetMemberStatus(source, 'combat', true)
SetTimeout(5000, function()
    exports['avalon_party']:SetMemberStatus(source, 'combat', false)
end)

    SetPartyBucket(source, bucket) → boolean

    将整个队伍移入路由桶（私人副本），适用于地牢、活动等。

    RegisterNetEvent('dungeon:enter', function(id)
    local ok = exports['avalon_party']:SetPartyBucket(source, 1000 + id)
    if not ok then
        TriggerClientEvent('ox_lib:notify', source, { type='error', description='需要队伍才能进入副本！' })
    end
end)

-- 退出后返回主世界
RegisterNetEvent('dungeon:exit', function()
    exports['avalon_party']:SetPartyBucket(source, 0)
end)

    GetPartyBucket(source) → number

    local bucket = exports['avalon_party']:GetPartyBucket(source)
print("队伍当前在桶：" .. bucket)

    客户端

    RequestMemberPick(mode, callbackEvent)

    激活成员选择模式：点击队伍成员进行选择。回调接收 { source, name }。

    RegisterCommand('治疗队友', function()
    exports['avalon_party']:RequestMemberPick('self', 'healing:target')
end)

RegisterNetEvent('healing:target', function(member)
    if not member then return end
    TriggerServerEvent('healing:apply', member.source, 50)
end)

    CancelMemberPick()

    exports['avalon_party']:CancelMemberPick()

    GetPartyTarget() → { ped, netId, source, isPlayer } | nil

    返回当前 TAB/CTRL 选择的目标信息。

    local target = exports['avalon_party']:GetPartyTarget()
if target and target.isPlayer then
    TriggerServerEvent('combat:attack', target.source)
end
